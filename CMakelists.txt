cmake_minimum_required(VERSION 3.6)

project(bmpc VERSION 1.0.0)

add_subdirectory(bmp)
add_subdirectory(color)

add_library(${PROJECT_NAME} STATIC bmpc.c)

target_link_libraries(${PROJECT_NAME} bmp)
target_link_libraries(${PROJECT_NAME} color)

INCLUDE_DIRECTORIES(bmp color .)

install(TARGETS ${PROJECT_NAME} DESTINATION bin)
install(FILES bmpc.h DESTINATION include)

add_subdirectory(examples)
